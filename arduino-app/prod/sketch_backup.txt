/*
 * sketch.ino — Production Dual IR Sensor Tracker via CD74HC4067 MUX
 * ==============================================================================
 *
 * This production sketch runs on the STM32U585 microcontroller of the Arduino
 * Uno Q. It interfaces with TWO Smartelex 5-Channel Tracker Sensors
 * (TCRT5000-based reflection sensors) through a single SmartElex CD74HC4067
 * 16-channel analog/digital multiplexer (MUX) breakout board.
 *
 * Instead of reading each sensor's 5 analog outputs directly from 5 dedicated
 * Arduino analog pins (as in the IR-sensor-test), all 10 sensor lines
 * (5 per sensor × 2 sensors = 10 channels) are routed through the MUX's
 * 16 input channels (C0–C15), and a single Arduino analog pin (SIG) reads
 * the selected channel's voltage.
 *
 * The MUX channel is selected by writing a 4-bit binary address to digital
 * pins S0–S3 on the Arduino. The CD74HC4067 internally connects the addressed
 * channel input to the common SIG output line.
 *
 * Technical Specifications & Design Concepts:
 * -------------------------------------------
 * 1. SmartElex CD74HC4067 16-Channel MUX Breakout:
 *    - IC: Texas Instruments CD74HC4067 (CMOS analog multiplexer/demultiplexer).
 *    - Channels: 16 independent analog/digital I/O channels (C0–C15).
 *    - Control Interface: 4 digital address lines (S0, S1, S2, S3) form a
 *      4-bit binary selector. The binary value on these lines determines which
 *      of the 16 channels is routed to the common SIG pin.
 *    - EN (Enable): Active-LOW. Must be tied to GND to enable the chip.
 *      When HIGH, all channels are disconnected (high-impedance).
 *    - Signal Path: Bidirectional. The selected Cx channel is electrically
 *      connected to SIG with very low on-resistance (~70Ω typ at 5V VCC).
 *    - Supply Voltage: 2V–6V. We power from 5V rail.
 *    - Channel Switching Time: ~12ns typical (negligible vs ADC sample time).
 *
 * 2. Smartelex 5-Channel Tracker Sensor (×2):
 *    - Same TCRT5000-based sensor as in IR-sensor-test.
 *    - Higher Reflection (ball present) → lower analog voltage.
 *    - Lower Reflection (empty pitch) → higher analog voltage.
 *
 * 3. MUX Channel Mapping:
 *    - Sensor 1 (5 channels): C0=IR1_S1, C1=IR2_S1, C2=IR3_S1, C3=IR4_S1, C4=IR5_S1
 *    - Sensor 2 (5 channels): C5=IR1_S2, C6=IR2_S2, C7=IR3_S2, C8=IR4_S2, C9=IR5_S2
 *    - Channels C10–C15 are unused (floating). We never address them.
 *
 * 4. Wiring Interface (MUX → Arduino Uno Q):
 *    - VCC  → 5V Pin (powers the MUX IC and both sensor modules)
 *    - GND  → GND Pin (common ground reference)
 *    - EN   → GND Pin (active-LOW enable; permanently enabled)
 *    - S0   → Digital Pin 2 (address bit 0, LSB)
 *    - S1   → Digital Pin 3 (address bit 1)
 *    - S2   → Digital Pin 4 (address bit 2)
 *    - S3   → Digital Pin 5 (address bit 3, MSB)
 *    - SIG  → Analog Pin A0 (common analog signal output from MUX)
 *
 * 5. Wiring Interface (Sensors → MUX):
 *    - Sensor 1 VCC  → 5V Pin (shared power rail)
 *    - Sensor 1 GND  → GND Pin (shared ground)
 *    - Sensor 1 IR1  → MUX C0
 *    - Sensor 1 IR2  → MUX C1
 *    - Sensor 1 IR3  → MUX C2
 *    - Sensor 1 IR4  → MUX C3
 *    - Sensor 1 IR5  → MUX C4
 *    - Sensor 2 VCC  → 5V Pin (shared power rail)
 *    - Sensor 2 GND  → GND Pin (shared ground)
 *    - Sensor 2 IR1  → MUX C5
 *    - Sensor 2 IR2  → MUX C6
 *    - Sensor 2 IR3  → MUX C7
 *    - Sensor 2 IR4  → MUX C8
 *    - Sensor 2 IR5  → MUX C9
 *
 * 6. Signal Processing Pipeline (per channel, identical to IR-sensor-test):
 *    - Moving average filter (window size 1 = passthrough for zero group delay)
 *    - Adaptive noise-based dynamic thresholding via EMA on sub-threshold deviations
 *    - Schmitt trigger hysteresis (trigger threshold = 3× noise, release = 1.5× noise)
 *    - Sub-threshold centered baseline adaptation (slow 0.999/0.001 EMA)
 *    - Fast recovery adaptation (0.9/0.1 EMA) for environmental shifts
 *    - Fast downward adaptation for permanent obstructions (>500ms shadow latch-up)
 *    - Zero-latency rising edge detection
 *    - Event aggregation with 500ms cooldown to coalesce multi-sensor ball transits
 *
 * 7. Output Format:
 *    When both sensors detect a ball transit, the output is a JSON-style array
 *    of two 5-element arrays:
 *      [[1,0,0,1,0],[0,0,1,0,0]]
 *    Each inner array represents one sensor's 5-channel trigger state
 *    (1 = beam broken by ball, 0 = no detection).
 *    When only one sensor detects, the other sensor's array will be all zeros.
 *
 * 8. Modular Architecture:
 *    - CD74HC4067Mux class: Encapsulates MUX address pin control and channel selection.
 *    - MuxAnalogTracker class: Encapsulates per-sensor analog reading via MUX,
 *      with the same readRaw/getPercentage interface as FiveChannelAnalogTracker.
 *    - All adaptive thresholding and event logic is in the global scope, exactly
 *      mirroring the IR-sensor-test structure but indexed for 2 sensors × 5 channels.
 */

// Include the Arduino Router Bridge library to enable serial output routing via
// MPU RPC. This is required on the Uno Q to bridge the STM32U585 MCU's serial
// output to the App Lab console through the onboard MPU (Linux SBC).
#include <Arduino_RouterBridge.h>

// ==============================================================================
// HARDWARE CONFIGURATION CONSTANTS
// ==============================================================================

// Total number of IR sensor modules connected to the system.
// Each module is a Smartelex 5-Channel Tracker Sensor with 5 independent outputs.
const int NUM_SENSORS = 2;

// Number of individual IR channels per sensor module.
// The Smartelex 5-Channel Tracker Sensor has 5 TCRT5000 elements (IR1–IR5).
const int CHANNELS_PER_SENSOR = 5;

// Total number of analog channels routed through the MUX.
// 2 sensors × 5 channels each = 10 total channels on MUX inputs C0–C9.
const int TOTAL_CHANNELS = NUM_SENSORS * CHANNELS_PER_SENSOR;

// Digital pin assignments for the MUX address lines.
// These 4 pins form a 4-bit binary address (S0=LSB, S3=MSB) that selects
// which of the 16 MUX input channels (C0–C15) is routed to the SIG output.
// We MUST use the board pin macros D2, D3, D4, D5 rather than raw integers.
// On the Zephyr RTOS core of the Arduino Uno Q, raw integers do not match board headers,
// causing the GPIO configurations to toggle wrong internal STM32 pins while the headers float.
const int MUX_S0_PIN = D2;  // Address bit 0 (least significant bit)
const int MUX_S1_PIN = D3;  // Address bit 1
const int MUX_S2_PIN = D4;  // Address bit 2
const int MUX_S3_PIN = D5;  // Address bit 3 (most significant bit)

// Analog pin connected to the MUX's common SIG (signal) output.
// All 10 sensor channels are time-division multiplexed through this single pin.
// The MUX's internal CMOS switch connects the selected Cx input to SIG.
const int MUX_SIG_PIN = A0;

// MUX channel-to-sensor mapping table.
// Each sensor's 5 IR outputs are connected to consecutive MUX input channels.
// Sensor 1: IR1→C0, IR2→C1, IR3→C2, IR4→C3, IR5→C4
// Sensor 2: IR1→C5, IR2→C6, IR3→C7, IR4→C8, IR5→C9
// This 2D array maps [sensorIndex][channelIndex] → MUX channel number.
const int MUX_CHANNEL_MAP[NUM_SENSORS][CHANNELS_PER_SENSOR] = {
  {0, 1, 2, 3, 4},   // Sensor 1: MUX channels C0–C4
  {5, 6, 7, 8, 9}    // Sensor 2: MUX channels C5–C9
};

// ==============================================================================
// CD74HC4067Mux CLASS
// ==============================================================================

/**
 * CD74HC4067Mux Class
 * Encapsulates all operations for controlling the SmartElex CD74HC4067
 * 16-channel analog/digital multiplexer breakout board.
 *
 * The CD74HC4067 is a CMOS analog multiplexer that routes one of 16 input
 * channels to a single common output (SIG). Channel selection is performed
 * by writing a 4-bit binary address to the S0–S3 control pins.
 *
 * This class handles:
 * - GPIO pin configuration for the 4 address lines
 * - Channel selection via binary address decomposition
 * - Analog reading from the SIG pin with proper settling time
 *
 * The EN (enable) pin is assumed to be hardwired to GND (active-LOW),
 * permanently enabling the MUX. No software control of EN is needed.
 */
class CD74HC4067Mux {
private:
  // Store the 4 digital pin numbers assigned to the MUX address lines.
  // s0Pin is the LSB, s3Pin is the MSB of the 4-bit channel address.
  int s0Pin, s1Pin, s2Pin, s3Pin;

  // Store the analog pin number connected to the MUX's SIG (signal) output.
  // This is the single analog input through which all 16 channels are read.
  int sigPin;

public:
  /**
   * Constructor — Stores pin assignments for the MUX control and signal lines.
   * Does NOT configure GPIO modes; that is deferred to begin() to follow
   * the Arduino two-phase initialization pattern (construct → setup → use).
   *
   * @param s0 Digital pin for MUX address bit 0 (S0, LSB)
   * @param s1 Digital pin for MUX address bit 1 (S1)
   * @param s2 Digital pin for MUX address bit 2 (S2)
   * @param s3 Digital pin for MUX address bit 3 (S3, MSB)
   * @param sig Analog pin for MUX common signal output (SIG)
   */
  CD74HC4067Mux(int s0, int s1, int s2, int s3, int sig) {
    s0Pin = s0;  // Map address bit 0 pin to instance storage.
    s1Pin = s1;  // Map address bit 1 pin to instance storage.
    s2Pin = s2;  // Map address bit 2 pin to instance storage.
    s3Pin = s3;  // Map address bit 3 pin to instance storage.
    sigPin = sig; // Map SIG analog pin to instance storage.
  }

  /**
   * begin — Configures GPIO pin modes for the MUX interface.
   * Address pins S0–S3 are set to OUTPUT mode (we drive them to select channels).
   * SIG pin is set to INPUT mode for analogRead operations.
   * Must be called once during setup() before any channel reads.
   */
  void begin() {
    // Configure address line pins as digital outputs.
    // These will be driven HIGH/LOW to form the binary channel address.
    pinMode(s0Pin, OUTPUT);
    pinMode(s1Pin, OUTPUT);
    pinMode(s2Pin, OUTPUT);
    pinMode(s3Pin, OUTPUT);

    // Configure the SIG pin as analog input.
    // Setting INPUT mode prevents internal pull-ups from interfering with
    // the analog voltage level passed through the MUX's CMOS switch.
    pinMode(sigPin, INPUT);
  }

  /**
   * selectChannel — Sets the 4-bit binary address on S0–S3 to route the
   * specified MUX channel to the SIG output.
   *
   * The CD74HC4067 interprets the 4 address pins as a binary number (0–15):
   *   channel = (S3 << 3) | (S2 << 2) | (S1 << 1) | (S0 << 0)
   *
   * We decompose the channel number using bitwise AND and right-shift
   * operations to extract each bit, then write it to the corresponding pin.
   *
   * Channel switching time is ~12ns (CD74HC4067 datasheet), but the analog
   * signal needs additional time to settle through the MUX's on-resistance
   * (~70Ω) and the ADC's internal sample-and-hold capacitor (~14pF on STM32).
   *
   * @param channel MUX channel number (0–15). Only 0–9 are used in this app.
   */
  void selectChannel(int channel) {
    // Extract bit 0 (LSB) from the channel number and write to S0.
    // The bitwise AND with 1 isolates the least significant bit.
    digitalWrite(s0Pin, channel & 1);

    // Extract bit 1 by right-shifting 1 position and masking with 1.
    digitalWrite(s1Pin, (channel >> 1) & 1);

    // Extract bit 2 by right-shifting 2 positions and masking with 1.
    digitalWrite(s2Pin, (channel >> 2) & 1);

    // Extract bit 3 (MSB) by right-shifting 3 positions and masking with 1.
    digitalWrite(s3Pin, (channel >> 3) & 1);
  }

  /**
   * readChannel — Reads the analog voltage from a specific MUX channel.
   *
   * WHY TWO DUMMY READS ARE REQUIRED FOR MUX-BASED READING:
   * -------------------------------------------------------
   * In the IR-sensor-test, each sensor has its own dedicated analog pin.
   * The Arduino's internal ADC MUX switches between well-matched internal
   * channels with minimal crosstalk. A single dummy read + 50μs is enough.
   *
   * With the external CD74HC4067 MUX, the situation is fundamentally different:
   * - All 10 sensor channels share a SINGLE analog pin (A0/SIG).
   * - When the MUX switches channels, the CMOS transmission gate injects
   *   ~10pC of charge (CD74HC4067 datasheet: charge injection spec) into
   *   the SIG line, perturbing the voltage.
   * - The ADC's sample-and-hold (S&H) capacitor retains residual charge
   *   from the PREVIOUSLY read channel's voltage level.
   * - The signal must propagate through the sensor's output impedance,
   *   the MUX's on-resistance (~70Ω), and into the S&H cap.
   *
   * A single dummy read partially converges the S&H cap toward the new
   * channel's true voltage, but leaves a systematic error of ~30-100+ ADC
   * counts. This error exceeds the adaptive thresholding system's minimum
   * trigger threshold floor (30 counts), causing false positive triggers.
   * The false triggers then activate the baseline's fast-adaptation pathways
   * (both downward latch-up clearance and upward recovery), creating a
   * positive-feedback oscillation that amplifies the initial MUX noise into
   * full-scale (0-1023) baseline swings.
   *
   * The fix is a TWO-STAGE dummy read convergence:
   * - Dummy Read 1: Flushes the bulk of the residual charge from the
   *   previous channel. The S&H cap converges ~90% toward the true voltage.
   * - 50μs settling: Allows any remaining charge redistribution.
   * - Dummy Read 2: Refines the S&H cap to within ~1-3 ADC counts of the
   *   true voltage (well below the 30-count threshold floor).
   * - 50μs settling: Final precision settling.
   * - Real Read: Accurate to within the ADC's inherent quantization noise.
   *
   * This two-stage approach matches the per-read accuracy of direct analog
   * pin reads (IR-sensor-test), ensuring the adaptive thresholding system
   * operates identically — no false triggers, no baseline oscillation.
   *
   * TIMING IMPACT:
   * - Per-channel read time: ~500μs (vs ~270μs with single dummy read)
   * - Full 10-channel scan: ~5ms (vs ~2.7ms)
   * - Effective scan rate: ~200 Hz (vs ~370 Hz)
   * - Still captures ball transits (~25ms) with ~5 samples per transit.
   *
   * @param channel MUX channel number (0–15). Only 0–9 should be used.
   * @return int Raw ADC value (0–1023 in 10-bit mode).
   */
  int readChannel(int channel) {
    // Bounds check to prevent addressing non-existent MUX channels.
    // The CD74HC4067 has 16 channels (0–15); we only use 0–9.
    if (channel < 0 || channel > 15) {
      return 0; // Return 0 for out-of-bounds to avoid undefined behavior.
    }

    // Step 1: Set the MUX address lines to select the target channel.
    // The CD74HC4067 routes the selected Cx input to the SIG output.
    selectChannel(channel);

    // Step 2: Wait 25 microseconds for the CMOS switch to fully settle.
    // The CD74HC4067 switching time is ~12ns, but we allow 25μs for the high
    // source impedance of the phototransistor pull-up (10kΩ) to charge the
    // parasitic capacitance of the MUX and SIG line (approx 65pF).
    // RC time constant: 10k * 65pF = 650ns. 25μs is >38 time constants (well settled).
    delayMicroseconds(25);

    // Step 3: Perform oversampling by taking 4 consecutive analog readings
    // and averaging them. This averages out high-frequency digital noise
    // coupled from the RPC communication/Bridge bus, and reduces the effective
    // source impedance, providing a stable, high-fidelity voltage representation.
    long sum = 0;
    for (int i = 0; i < 4; i++) {
      sum += analogRead(sigPin);
      // Small 5μs delay between oversamples to allow high-frequency power supply
      // transient noise to redistribute.
      delayMicroseconds(5);
    }

    // Return the averaged ADC value (10-bit resolution).
    return (int)(sum / 4);
  }
};

// ==============================================================================
// MuxAnalogTracker CLASS
// ==============================================================================

/**
 * MuxAnalogTracker Class
 * Provides the same readRaw/getPercentage interface as FiveChannelAnalogTracker
 * from IR-sensor-test, but reads sensor channels through the CD74HC4067 MUX
 * instead of directly from Arduino analog pins.
 *
 * Each instance represents one 5-channel IR sensor module. It stores the
 * MUX channel numbers for its 5 IR outputs and delegates reads to the
 * shared CD74HC4067Mux instance.
 *
 * This abstraction allows the signal processing pipeline (baselines, triggers,
 * thresholds) to interact with the sensor through the same API regardless of
 * whether it's connected directly or through a MUX.
 */
class MuxAnalogTracker {
private:
  // Pointer to the shared CD74HC4067Mux instance that controls the physical MUX.
  // Multiple MuxAnalogTracker instances (one per sensor) share the same MUX object.
  CD74HC4067Mux* mux;

  // Array of 5 MUX channel numbers that this sensor's IR outputs are connected to.
  // For Sensor 1: {0, 1, 2, 3, 4}. For Sensor 2: {5, 6, 7, 8, 9}.
  int muxChannels[CHANNELS_PER_SENSOR];

  // The maximum possible reading from the Analog-to-Digital Converter.
  // By default, the UNO Q's ADC runs in 10-bit mode, giving a maximum value of
  // 1023. This is used for percentage calculations (identical to IR-sensor-test).
  int adcMaxVal;

public:
  /**
   * Constructor — Associates this tracker with a MUX and maps its 5 channels.
   *
   * @param muxPtr Pointer to the shared CD74HC4067Mux instance.
   * @param channels Array of 5 MUX channel numbers for this sensor's IR1–IR5.
   */
  MuxAnalogTracker(CD74HC4067Mux* muxPtr, const int channels[CHANNELS_PER_SENSOR]) {
    mux = muxPtr; // Store the shared MUX reference.
    // Copy the 5 channel numbers from the provided array into local storage.
    for (int i = 0; i < CHANNELS_PER_SENSOR; i++) {
      muxChannels[i] = channels[i];
    }
    adcMaxVal = 1023; // Default maximum ADC reading for a 10-bit conversion.
  }

  /**
   * readRaw — Reads the raw analog state of a specific sensor channel (0–4)
   * by routing through the MUX.
   *
   * This method maps the logical sensor channel index (0=IR1 leftmost,
   * 4=IR5 rightmost) to the physical MUX channel number, then delegates
   * to the MUX's readChannel() method which handles all settling and
   * dummy-read logic.
   *
   * @param index 0-based sensor channel index (0 = IR1 leftmost, 4 = IR5 rightmost).
   * @return int Raw ADC value (typically 0–1023).
   */
  int readRaw(int index) {
    // Bounds checking to avoid out-of-bounds memory reading of channel array.
    if (index < 0 || index >= CHANNELS_PER_SENSOR) {
      return 0; // Return 0 if out of bounds to avoid memory fault.
    }
    // Delegate to the MUX, using the mapped channel number for this sensor index.
    return mux->readChannel(muxChannels[index]);
  }

  /**
   * getPercentageFromRaw — Calculates reflection percentage for a given raw ADC value.
   * 100% means highest reflection (sensor close to highly reflective surface / ball).
   * 0% means lowest reflection (sensor over non-reflective surface or dark line).
   * Since lower analog values indicate higher reflection (TCRT5000 physics), we
   * invert the percentage calculation.
   *
   * This is identical to FiveChannelAnalogTracker::getPercentageFromRaw() in IR-sensor-test.
   *
   * @param rawVal The raw ADC value (typically 0–1023).
   * @return float The computed reflection percentage (0.0–100.0).
   */
  float getPercentageFromRaw(int rawVal) {
    // Map raw value to fraction of ADC max: 0.0 (maximum reflection) to 1.0
    // (minimum reflection). Cast rawVal to float to avoid integer division
    // truncation before performing division.
    float fraction = (float)rawVal / adcMaxVal;
    // Invert fraction (so 0V/0 reading becomes 100% reflection, 1023 becomes 0%).
    float percentage = (1.0 - fraction) * 100.0;
    return percentage;
  }

  /**
   * getPercentage — Calculates reflection percentage for a channel by reading fresh data.
   * Combines readRaw and getPercentageFromRaw into a single convenience method.
   *
   * @param index 0-based sensor channel index (0–4).
   * @return float Reflection percentage (0.0–100.0).
   */
  float getPercentage(int index) {
    if (index < 0 || index >= CHANNELS_PER_SENSOR) {
      return 0.0;
    }
    int rawVal = readRaw(index);
    return getPercentageFromRaw(rawVal);
  }
};

// Global instance of the CD74HC4067 MUX controller.
// Configured with S0=D2, S1=D3, S2=D4, S3=D5, SIG=A0.
// This single MUX object is shared by both sensor tracker instances.
CD74HC4067Mux mux(MUX_S0_PIN, MUX_S1_PIN, MUX_S2_PIN, MUX_S3_PIN, MUX_SIG_PIN);

// Global instances of MuxAnalogTracker, one per IR sensor module.
// Each tracker is initialized with a pointer to the shared MUX and
// its specific MUX channel mapping from the MUX_CHANNEL_MAP table.
MuxAnalogTracker tracker1(&mux, MUX_CHANNEL_MAP[0]); // Sensor 1: MUX C0–C4
MuxAnalogTracker tracker2(&mux, MUX_CHANNEL_MAP[1]); // Sensor 2: MUX C5–C9

// Convenience array of pointers to both trackers for indexed iteration.
// This allows the main loop to process both sensors using a single for-loop
// over sensorIndex, rather than duplicating code for each sensor.
MuxAnalogTracker* trackers[NUM_SENSORS] = {&tracker1, &tracker2};

// ==============================================================================
// ADAPTIVE THRESHOLDING & SIGNAL PROCESSING STATE ARRAYS
// ==============================================================================
// All arrays are dimensioned [NUM_SENSORS][CHANNELS_PER_SENSOR] to hold
// per-sensor, per-channel state. This is the exact same set of state variables
// as IR-sensor-test, but expanded from 1D (5 channels) to 2D (2×5 channels).

// [Static Thresholds Removed]
// We use Adaptive Noise-Based Dynamic Thresholding, identical to IR-sensor-test.
// Each sensor channel continuously calculates its own ambient electrical/lighting
// noise floor (MAD via EMA) and mathematically generates perfectly scaled triggers.

// No artificial polling delay. Loop runs at maximum physical speed (~1500–2000 Hz
// per full scan of all 10 channels) to never miss a fast-moving ball transit.

// Moving average window size. Set to 1 to eliminate group delay and capture
// instantaneous ball transits. This makes the filter a direct passthrough,
// which is the proven configuration from IR-sensor-test.
const int MOVING_AVERAGE_WINDOW_SIZE = 1;

// Circular buffer to store raw ADC histories for calculating moving averages.
// Indexed as [sensorIndex][channelIndex][sampleIndex].
int rawReadingsHistory[NUM_SENSORS][CHANNELS_PER_SENSOR][MOVING_AVERAGE_WINDOW_SIZE];

// Write index for the circular buffer. Shared across all channels since
// all channels are sampled synchronously in each loop iteration.
// Ranges from 0 to (MOVING_AVERAGE_WINDOW_SIZE - 1) and wraps around.
int historyWriteIndex = 0;

// Calibration baseline arrays — the reference "empty pitch" ADC values captured
// at startup. Integer version for fast runtime comparisons.
int calibrationBaseline[NUM_SENSORS][CHANNELS_PER_SENSOR];

// Calibration baseline as floating-point for precise long-term drift tracking.
// Using float (32-bit IEEE 754) prevents truncation during slow EMA updates.
// This is identical to calibrationBaselineFloat in IR-sensor-test.
float calibrationBaselineFloat[NUM_SENSORS][CHANNELS_PER_SENSOR];

// Previous trigger state for each channel. Required for Schmitt trigger
// state-transition detection (rising/falling edge detection).
bool prevTriggerState[NUM_SENSORS][CHANNELS_PER_SENSOR];

// Recovery state flag. When the baseline falls significantly behind the signal
// (e.g., due to thermal drift or shadow clearing), this flag enables fast
// upward adaptation to re-center the baseline.
bool isRecovering[NUM_SENSORS][CHANNELS_PER_SENSOR];

// Timestamp (in milliseconds) when each channel entered the triggered state.
// Used to calculate transit duration and filter out slow entities (e.g., hands).
unsigned long triggerStartTime[NUM_SENSORS][CHANNELS_PER_SENSOR];

// Peak reflection tracker — stores the lowest raw ADC value observed during
// a trigger event (lowest raw = strongest reflection = closest ball pass).
int minRawDuringTrigger[NUM_SENSORS][CHANNELS_PER_SENSOR];

// Filtered (moving-averaged) readings. Float type prevents rounding error
// accumulation across averaging iterations.
float filteredReadings[NUM_SENSORS][CHANNELS_PER_SENSOR];

// Previous iteration's filtered readings, used for derivative calculations
// and rate-of-change analysis in future enhancements.
float prevFilteredReadings[NUM_SENSORS][CHANNELS_PER_SENSOR];

// Noise variance (noise floor) tracked via Exponential Moving Average (EMA).
// This represents the ambient electrical and optical noise amplitude for each
// channel. The dynamic thresholds are mathematically derived from this value.
// Initialized to 5.0 (same as IR-sensor-test) as a conservative starting estimate.
float noiseVarianceFloat[NUM_SENSORS][CHANNELS_PER_SENSOR];

// Dynamically calculated trigger thresholds (integer, for fast comparison).
// Trigger threshold = max(30, noiseVarianceFloat × 3.0).
// This is the Schmitt trigger HIGH threshold — deviation must exceed this
// to transition from untriggered to triggered state.
int dynamicTriggerThresholds[NUM_SENSORS][CHANNELS_PER_SENSOR];

// Dynamically calculated release thresholds (integer, for fast comparison).
// Release threshold = max(20, noiseVarianceFloat × 1.5).
// This is the Schmitt trigger LOW threshold — deviation must drop below this
// to transition from triggered to untriggered state.
// The gap between trigger and release provides hysteresis to prevent chattering.
int dynamicReleaseThresholds[NUM_SENSORS][CHANNELS_PER_SENSOR];

// Global event aggregation flags — per sensor, per channel.
// When a sensor channel fires a rising edge, its flag is set to true.
// All flags are coalesced into a single output array when any trigger fires.
bool sensorsTriggeredInEvent[NUM_SENSORS][CHANNELS_PER_SENSOR];

// Cooldown timer to prevent multiple prints for the same ball transit.
// After printing an event array, we engage a 500ms blind period during which
// no new events are printed (but rising edges are still tracked for baseline logic).
unsigned long lastEventPrintTime = 0;

// Cooldown duration in milliseconds. 500ms is the proven value from IR-sensor-test
// that prevents a single ball pass from generating multiple output arrays
// as it rolls across multiple sensors.
const unsigned long EVENT_COOLDOWN_MS = 500;

// ==============================================================================
// INITIALIZATION HELPER FUNCTIONS
// ==============================================================================

/**
 * initializeStateArrays — Zeros out all per-sensor, per-channel state arrays
 * to known starting values.
 *
 * This is called once during setup() to ensure all adaptive thresholding,
 * baseline tracking, and event aggregation variables start in a clean state.
 * The initial values match those used in IR-sensor-test:
 * - Baselines: 0 (will be calibrated during startup)
 * - Trigger states: false (untriggered)
 * - Noise variance: 5.0 (conservative starting estimate)
 * - Dynamic thresholds: 30/20 (minimum floor values)
 * - Min raw during trigger: 1023 (ADC max, will decrease on ball detection)
 */
void initializeStateArrays() {
  for (int s = 0; s < NUM_SENSORS; s++) {
    for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
      calibrationBaseline[s][c] = 0;
      calibrationBaselineFloat[s][c] = 0.0;
      prevTriggerState[s][c] = false;
      isRecovering[s][c] = false;
      triggerStartTime[s][c] = 0;
      minRawDuringTrigger[s][c] = 1023;
      filteredReadings[s][c] = 0.0;
      prevFilteredReadings[s][c] = 0.0;
      noiseVarianceFloat[s][c] = 5.0;
      dynamicTriggerThresholds[s][c] = 30;
      dynamicReleaseThresholds[s][c] = 20;
      sensorsTriggeredInEvent[s][c] = false;
    }
  }
}

// ==============================================================================
// SETUP
// ==============================================================================

/**
 * setup() — Microcontroller initialization routine.
 * Executed once when the board is booted or reset.
 *
 * Performs the following sequence:
 * 1. Initialize serial communication and Router Bridge for App Lab console output.
 * 2. Initialize the MUX GPIO pins (address lines as outputs, SIG as input).
 * 3. Initialize all state arrays to clean starting values.
 * 4. Wait for hardware stabilization (IR emitters, ADC reference, MUX).
 * 5. Seed the circular buffer with initial readings.
 * 6. Settle the buffer with 20 active read cycles.
 * 7. Calculate and store calibration baselines.
 * 8. Print diagnostic banner and wiring information.
 */
void setup() {
  // Initialize virtual serial transceiver (baud rate is ignored by RPC driver
  // but set to 9600 for App Lab UI compatibility).
  Serial.begin(9600);

  // Initialize the Router Bridge module to establish virtual serial interface
  // to App Lab. This is required for Serial.println() output to appear in
  // the App Lab console on the Arduino Uno Q.
  Bridge.begin();

  // Initialize the MUX's GPIO pins (S0–S3 as OUTPUT, SIG as INPUT).
  mux.begin();

  // Initialize all state arrays to clean starting values.
  initializeStateArrays();

  // Print pre-calibration notification to inform the operator that the system
  // is stabilizing and the pitch must be kept clear of objects.
  Serial.println("Stabilizing hardware components... Keep pitch clear!");

  // Pause for 2000 milliseconds to allow power rails and IR emitters to stabilize.
  // The Smartelex TCRT5000 GaAs diodes and NPN phototransistors need time to
  // reach thermal and optical equilibrium. The MUX's CMOS switches also need
  // VCC to stabilize for consistent on-resistance. The MCU's internal analog
  // reference voltage requires settling time for accurate ADC conversions.
  delay(2000);

  // Seed the circular buffer history with initial raw readings from each channel
  // on each sensor. This prevents startup transient spikes in the moving average.
  for (int s = 0; s < NUM_SENSORS; s++) {
    for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
      // Read the current raw ADC level from this sensor's channel via MUX.
      int initialVal = trackers[s]->readRaw(c);
      // Fill all window slots in the circular history buffer with the initial value.
      for (int sample = 0; sample < MOVING_AVERAGE_WINDOW_SIZE; sample++) {
        rawReadingsHistory[s][c][sample] = initialVal;
      }
    }
  }

  // Settle the circular history buffer by running the read loop for 50 cycles.
  // Increased from 20 cycles (IR-sensor-test) to 50 cycles because the MUX-based
  // signal path has higher per-read variance than direct analog pins. More settling
  // cycles allow the moving average buffer to converge to the true ambient level,
  // producing a more accurate calibration baseline. Each cycle reads all 10
  // channels (2 sensors × 5 channels) through the MUX.
  for (int sample = 0; sample < 50; sample++) {
    for (int s = 0; s < NUM_SENSORS; s++) {
      for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
        // Store the active raw reading in the buffer at the current write slot.
        rawReadingsHistory[s][c][historyWriteIndex] = trackers[s]->readRaw(c);
      }
    }
    // Increment the write index in circular fashion.
    historyWriteIndex = (historyWriteIndex + 1) % MOVING_AVERAGE_WINDOW_SIZE;
    // Wait 2 milliseconds to allow the RTOS and RouterBridge background tasks
    // to execute between cycles (same as IR-sensor-test and main loop yield).
    delay(2);
  }

  // Calculate the moving average of the history buffer to establish startup baselines.
  for (int s = 0; s < NUM_SENSORS; s++) {
    for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
      // Accumulator variable to hold the sum of the history buffer for this channel.
      long sum = 0;
      for (int sample = 0; sample < MOVING_AVERAGE_WINDOW_SIZE; sample++) {
        sum += rawReadingsHistory[s][c][sample];
      }
      // Calculate the mathematical average as a float.
      float average = (float)sum / MOVING_AVERAGE_WINDOW_SIZE;
      // Store the float baseline for precise drift tracking.
      calibrationBaselineFloat[s][c] = average;
      // Store the integer baseline for fast runtime comparisons.
      calibrationBaseline[s][c] = (int)average;
    }
  }

  // Print calibrated baseline readings for diagnostic verification.
  for (int s = 0; s < NUM_SENSORS; s++) {
    Serial.print("Sensor ");
    Serial.print(s + 1);
    Serial.print(" Baselines: ");
    for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
      Serial.print("IR");
      Serial.print(c + 1);
      Serial.print(":");
      Serial.print(calibrationBaseline[s][c]);
      Serial.print(" ");
    }
    Serial.println();
  }

  // Print introductory diagnostic banner to App Lab console.
  Serial.println("=====================================================");
  Serial.println("  PRODUCTION DUAL IR TRACKER via CD74HC4067 MUX");
  Serial.println("=====================================================");
  Serial.println("MUX Wiring (CD74HC4067 -> Arduino UNO Q):");
  Serial.println("  VCC (Power)    -> 5V Pin");
  Serial.println("  GND (Ground)   -> GND Pin");
  Serial.println("  EN  (Enable)   -> GND Pin (active-LOW, always ON)");
  Serial.println("  S0  (Addr b0)  -> Digital Pin 2");
  Serial.println("  S1  (Addr b1)  -> Digital Pin 3");
  Serial.println("  S2  (Addr b2)  -> Digital Pin 4");
  Serial.println("  S3  (Addr b3)  -> Digital Pin 5");
  Serial.println("  SIG (Signal)   -> Analog Pin A0");
  Serial.println("-----------------------------------------------------");
  Serial.println("Sensor 1 -> MUX: IR1=C0 IR2=C1 IR3=C2 IR4=C3 IR5=C4");
  Serial.println("Sensor 2 -> MUX: IR1=C5 IR2=C6 IR3=C7 IR4=C8 IR5=C9");
  Serial.println("=====================================================");
  Serial.println("Output: [[S1_IR1..S1_IR5],[S2_IR1..S2_IR5]]");
  Serial.println("=====================================================");
}

// ==============================================================================
// MAIN LOOP
// ==============================================================================

/**
 * loop() — Continuous application runtime routine.
 * Executed repeatedly after setup() completes.
 *
 * This is a direct translation of the IR-sensor-test loop() to operate on
 * 2 sensors × 5 channels = 10 total channels, all read through the MUX.
 *
 * The processing pipeline for each channel is identical:
 * 1. Read raw ADC value via MUX.
 * 2. Update circular buffer and compute moving average.
 * 3. Calculate signal deviation from calibrated baseline.
 * 4. Evaluate Schmitt trigger state using dynamic hysteresis thresholds.
 * 5. Update noise envelope EMA (sub-threshold positive deviations only).
 * 6. Recalculate dynamic trigger/release thresholds from noise envelope.
 * 7. Perform sub-threshold centered baseline adaptation.
 * 8. Zero-latency rising edge detection → event aggregation → output.
 */
void loop() {
  // Static variable to track the last time we printed a heartbeat diagnostic message.
  // Using 'static' preserves value across loop() invocations without global scope pollution.
  static unsigned long lastDebugPrint = 0;

  // Local arrays for current raw readings and trigger states.
  // Stack allocation for fast access during comparisons.
  int currentRawReadings[NUM_SENSORS][CHANNELS_PER_SENSOR];
  bool currentTriggerState[NUM_SENSORS][CHANNELS_PER_SENSOR];

  // ============================================================================
  // PHASE 1: READ, FILTER, AND EVALUATE ALL 10 CHANNELS
  // ============================================================================
  // Iterate over both sensors and all 5 channels per sensor.
  // Each channel is read from the MUX, filtered, and evaluated for trigger state.
  for (int s = 0; s < NUM_SENSORS; s++) {
    for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
      // --- READ ---
      // Query the MUX-based tracker to read the analog voltage for this channel.
      // The MUX switches to the appropriate Cx channel, settles, and reads.
      currentRawReadings[s][c] = trackers[s]->readRaw(c);

      // Write the new raw reading into the circular buffer at the current write index.
      rawReadingsHistory[s][c][historyWriteIndex] = currentRawReadings[s][c];

      // --- FILTER (MOVING AVERAGE) ---
      // Compute the sum of the history buffer for this channel.
      long sum = 0;
      for (int sample = 0; sample < MOVING_AVERAGE_WINDOW_SIZE; sample++) {
        sum += rawReadingsHistory[s][c][sample];
      }
      // Calculate the moving average as a float.
      // With MOVING_AVERAGE_WINDOW_SIZE=1, this is a direct passthrough.
      filteredReadings[s][c] = (float)sum / MOVING_AVERAGE_WINDOW_SIZE;

      // --- SIGNAL DEVIATION ---
      // Calculate how far the filtered reading has deviated from the baseline.
      // A positive diff means the sensor sees MORE reflection than ambient
      // (i.e., a ball is present, driving the voltage lower).
      int diff = calibrationBaseline[s][c] - (int)filteredReadings[s][c];

      // --- SCHMITT TRIGGER STATE EVALUATION ---
      // Evaluate state transition using dynamic hysteresis thresholds.
      // This is evaluated BEFORE noise tracking to ensure ball strikes are
      // mathematically excluded from the noise envelope (same as IR-sensor-test).
      bool newState = prevTriggerState[s][c];
      if (!prevTriggerState[s][c]) {
        // Currently untriggered: need diff >= triggerThreshold to transition to triggered.
        if (diff >= dynamicTriggerThresholds[s][c]) newState = true;
      } else {
        // Currently triggered: need diff < releaseThreshold to transition to untriggered.
        if (diff < dynamicReleaseThresholds[s][c]) newState = false;
      }
      currentTriggerState[s][c] = newState;

      // --- NOISE ENVELOPE TRACKING ---
      // Track pure ambient noise using Exponential Moving Average (EMA).
      // ONLY sub-threshold positive deviations are sampled. This ensures that
      // actual ball detections (which exceed the trigger threshold) do NOT
      // contaminate the noise estimate.
      // EMA coefficient: α=0.01 (slow tracking, 99% memory retention).
      if (diff > 0 && !currentTriggerState[s][c]) {
        noiseVarianceFloat[s][c] = (noiseVarianceFloat[s][c] * 0.99) + ((float)diff * 0.01);
      }

      // --- DYNAMIC THRESHOLD CALCULATION ---
      // Derive trigger and release thresholds from the learned noise envelope.
      // Trigger = max(30, noise × 3.0) — must exceed 3× noise to fire.
      // Release = max(20, noise × 1.5) — must drop below 1.5× noise to release.
      // The min values (30, 20) provide a hard floor to prevent false triggers
      // in extremely quiet environments.
      int dynamicTrigger = max(30, (int)(noiseVarianceFloat[s][c] * 3.0));
      int dynamicRelease = max(20, (int)(noiseVarianceFloat[s][c] * 1.5));
      dynamicTriggerThresholds[s][c] = dynamicTrigger;
      dynamicReleaseThresholds[s][c] = dynamicRelease;

      // --- SUB-THRESHOLD CENTERED BASELINE ADAPTATION ---
      // The baseline must slowly track environmental changes (temperature drift,
      // ambient light changes) while ignoring ball transits.
      int upwardDiff = filteredReadings[s][c] - calibrationBaselineFloat[s][c];

      // State Machine: If the baseline falls behind the signal (e.g., thermal
      // drift or shadow cleared), enter recovery state for fast adaptation.
      if (upwardDiff > dynamicTriggerThresholds[s][c]) {
        isRecovering[s][c] = true;
      }
      // Exit recovery once baseline has caught up (centered).
      if (upwardDiff <= 0) {
        isRecovering[s][c] = false;
      }

      if (isRecovering[s][c]) {
        // Fast upward adaptation (α=0.1) to recover from environmental shifts.
        // 90% memory, 10% new value — converges in ~23 samples (1/e time constant).
        calibrationBaselineFloat[s][c] = (calibrationBaselineFloat[s][c] * 0.9) + (filteredReadings[s][c] * 0.1);
      } else {
        if (currentTriggerState[s][c] && (millis() - triggerStartTime[s][c] > 500)) {
          // Fast downward adaptation to clear latch-up from permanent obstructions
          // (e.g., a hand resting on the sensor for >500ms).
          // Uses the same fast α=0.1 coefficient to quickly absorb the obstruction
          // into the baseline, allowing the sensor to return to normal operation.
          calibrationBaselineFloat[s][c] = (calibrationBaselineFloat[s][c] * 0.9) + (filteredReadings[s][c] * 0.1);
        } else {
          // Slow, symmetric adaptation (α=0.001) to keep the baseline perfectly
          // centered in the ambient noise wave. 99.9% memory retention ensures
          // ball transits (which are brief) barely affect the baseline.
          calibrationBaselineFloat[s][c] = (calibrationBaselineFloat[s][c] * 0.999) + (filteredReadings[s][c] * 0.001);
        }
      }

      // Sync the integer baseline from the float version for next iteration's comparisons.
      calibrationBaseline[s][c] = (int)calibrationBaselineFloat[s][c];
    }
  }

  // ============================================================================
  // PHASE 2: ZERO-LATENCY RISING EDGE DETECTION & EVENT AGGREGATION
  // ============================================================================
  // Trigger instantly upon beam break to satisfy the ~25ms physical ball transit
  // constraint. We must fire the exact millisecond any sensor is breached.

  bool triggerFiredThisLoop = false;

  // Only evaluate new triggers if past the 500ms cooldown from the last shot.
  // This prevents a single ball pass from printing multiple arrays as it
  // rolls over multiple sensors sequentially.
  if (millis() - lastEventPrintTime > EVENT_COOLDOWN_MS) {
    for (int s = 0; s < NUM_SENSORS; s++) {
      for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
        if (currentTriggerState[s][c]) {
          if (!prevTriggerState[s][c]) {
            // RISING EDGE detected: This sensor channel was breached this exact loop.
            // Record the timestamp for transit duration tracking and shadow latch-up logic.
            triggerStartTime[s][c] = millis();

            // Mark this channel as triggered in the event aggregation array.
            sensorsTriggeredInEvent[s][c] = true;

            // Flag that at least one trigger fired this loop iteration.
            triggerFiredThisLoop = true;
          }
        }
      }
    }

    // If ANY sensor channel was breached during this loop iteration, print the
    // ML-ready output array immediately. 0ms delay. Maximum physical speed.
    // Output format: [[S1_IR1,S1_IR2,S1_IR3,S1_IR4,S1_IR5],[S2_IR1,...,S2_IR5]]
    if (triggerFiredThisLoop) {
      Serial.print("[[");
      // Print Sensor 1's 5-element trigger array.
      for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
        Serial.print(sensorsTriggeredInEvent[0][c] ? "1" : "0");
        if (c < CHANNELS_PER_SENSOR - 1) Serial.print(",");
        // Reset the flag immediately after printing.
        sensorsTriggeredInEvent[0][c] = false;
      }
      Serial.print("],[");
      // Print Sensor 2's 5-element trigger array.
      for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
        Serial.print(sensorsTriggeredInEvent[1][c] ? "1" : "0");
        if (c < CHANNELS_PER_SENSOR - 1) Serial.print(",");
        // Reset the flag immediately after printing.
        sensorsTriggeredInEvent[1][c] = false;
      }
      Serial.println("]]");

      // Engage the 500ms cooldown to prevent spam from a single ball transit.
      lastEventPrintTime = millis();
    }
  } else {
    // During cooldown: still record trigger start times for baseline adaptation.
    // The 500ms shadow latch-up clearance logic in baseline adaptation needs
    // accurate triggerStartTime values even during cooldown.
    for (int s = 0; s < NUM_SENSORS; s++) {
      for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
        if (currentTriggerState[s][c] && !prevTriggerState[s][c]) {
          triggerStartTime[s][c] = millis();
        }
      }
    }
  }

  // ============================================================================
  // PHASE 3: DIAGNOSTIC HEARTBEAT
  // ============================================================================
  // Non-blocking 1-second diagnostic heartbeat to output live state.
  // This allows the operator to verify sensor health, baseline tracking,
  // and dynamic threshold adaptation in real-time without triggering events.
  if (millis() - lastDebugPrint > 1000) {
    lastDebugPrint = millis();
    for (int s = 0; s < NUM_SENSORS; s++) {
      Serial.print("HB S");
      Serial.print(s + 1);
      Serial.print(" | BL:[");
      for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
        Serial.print(calibrationBaseline[s][c]);
        if (c < CHANNELS_PER_SENSOR - 1) Serial.print(",");
      }
      Serial.print("] | Raw:[");
      for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
        Serial.print(currentRawReadings[s][c]);
        if (c < CHANNELS_PER_SENSOR - 1) Serial.print(",");
      }
      Serial.print("] | DT:[");
      for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
        Serial.print(dynamicTriggerThresholds[s][c]);
        if (c < CHANNELS_PER_SENSOR - 1) Serial.print(",");
      }
      Serial.println("]");
    }
  }

  // ============================================================================
  // PHASE 4: STATE BUFFER UPDATE
  // ============================================================================
  // Copy current trigger states to the previous-state buffer for edge detection
  // on the next cycle. Also preserve filtered readings for derivative analysis.
  for (int s = 0; s < NUM_SENSORS; s++) {
    for (int c = 0; c < CHANNELS_PER_SENSOR; c++) {
      prevTriggerState[s][c] = currentTriggerState[s][c];
      prevFilteredReadings[s][c] = filteredReadings[s][c];
    }
  }

  // Increment the circular buffer write index for the next loop iteration.
  // Wraps around using modulo to maintain circular buffer boundary logic.
  historyWriteIndex = (historyWriteIndex + 1) % MOVING_AVERAGE_WINDOW_SIZE;

  // Yield 2ms at the end of the loop to allow the underlying RTOS and
  // RouterBridge to flush the virtual serial buffer. Without this delay,
  // the tight loop starves the background RPC tasks, crashing the bridge
  // and preventing ANY serial prints from reaching the console.
  // This is the exact same critical delay from IR-sensor-test.
  delay(2);
}
